00020 /* I needed a hashtable for a project and wanted to code my own. This
00021  * hashtable compromises speed for reliability, specifically with
00022  * growing the hashtable.
00023  *
00024  * The hashtable does not grow automatically, but when the hashtable
00025  * grow function is called. Growing the hashtable is a safe operation:
00026  * if growing the hashtable fails, the existing hashtable is not
00027  * destroyed or modified.
00028  *
00029  * This hashtable is not thread-safe.
00030  */
00031

00043   void *value;                  /* value for this node */
00044   size_t vallen;                /* length of the value */

00049 typedef struct hashtab_t

00053   int count;                    /* number if items in this table */

00076 /* Initialize a new hashtable (set bookingkeeping data) and return a
00077  * pointer to the hashtable. A hash function may be provided. If no
00078  * function pointer is given (a NULL pointer), then the built in hash
00079  * function is used. A NULL pointer returned if the creation of the
00080  * hashtable failed due to a failed malloc(). */

00082                     int (*hash_func)

00098 /* Change the size of the hashtable. It will allocate a new hashtable
